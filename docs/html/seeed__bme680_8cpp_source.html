247 static uint8_t spi_transfer(uint8_t x)
248 {
249  if (0==user_define_spi_pinmap_flag)
250  {
251  return SPI.transfer(x);
252  }

254  uint8_t reply = 0;
255  for (int i=7; i>=0; i--)
256  {
257  reply <<= 1;
258  digitalWrite(spi_pinmap_sck, LOW);
259  digitalWrite(spi_pinmap_mosi, x & (1<<i));
260  digitalWrite(spi_pinmap_sck, HIGH);

262  {
263  reply |= 1;
264  }
265  }
266  return reply;
